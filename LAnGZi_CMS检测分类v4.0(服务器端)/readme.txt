---------------配置文件说明---------------------


[Server]
host = 127.0.0.1
# 本地填写127.0.0.1，远程IP填写自己的
username = root
password = root
db = lang_cms_find
# 数据库名
port = 3306
# 端口

[Config]
thread_s = 20
# 线程数
level = 2
# 扫描检测等级，1-3，默认为2

# level 1 : 根据网页内容，请求头内容，robots文件判断
# level 2 : 根据网页内容，请求头内容，robots文件,网址内容，网页文件识别判断
# level 3 : 根据网页内容，请求头内容，robots文件,网址内容，网页文件,根据域名生成路径识别判断,速度太慢

check_waf = 1
#检测是否存在WAF，如果存在WAF就跳过 1yes 0no

check_cms = 1
# 检测cms类型，设置0则不检测cms类型，只无限采集url，设置1则采集url，识别cms
